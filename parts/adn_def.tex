\begin{itemize}
  \item The set of components: $N = \{ a, b, c, \dots \}$
  \item The local states (of one component $a$): $x^a \in \F^a = \segm{0}{l_a}$
  \item The global states (of all components): $\F = \F^a \times \F^b \times \F^c \times \dots$  % \underset{a \in N}{\times}
  \item The evolution function (of one component $a$): $f^a : \F \rightarrow \F^a$
\end{itemize}
